Torque do motor à força de tração

\begin{eqnarray} \nonumber \mathrm{T}_{\mathrm{c}}&=&\mathrm{T}_{\mathrm{e}}-\mathrm{I}_{\mathrm{e}} \alpha_{\mathrm{e}} \\ \nonumber \mathrm{T}_{\mathrm{d}}&=&\left(\mathrm{T}_{\mathrm{c}}-\mathrm{I}_{\mathrm{t}} \alpha_{\mathrm{e}}\right) \mathrm{N}_{\mathrm{t}} \\ \nonumber \mathrm{T}_{\mathrm{a}} &=&\left(\mathrm{T}_{\mathrm{d}}-\mathrm{I}_{\mathrm{d}} \alpha_{\mathrm{d}}\right) \mathrm{N}_{\mathrm{f}} \\ \nonumber \mathrm{T}_{\mathrm{a}}&=&\mathrm{F}_{\mathrm{X}} \mathrm{r}+\mathrm{I}_{\mathrm{W}} \alpha_{\mathrm{W}}\\ \end{eqnarray}

\begin{eqnarray} \nonumber \alpha_{\mathrm{d}}&=&\mathrm{N}_{\mathrm{f}} \alpha_{\mathrm{w}} \\ \nonumber \alpha_{\mathrm{e}}&=&\mathrm{N}_{\mathrm{t}} \alpha_{\mathrm{d}} \\ \nonumber \alpha_{\mathrm{e}}&=&\mathrm{N}_{\mathrm{t}} \mathrm{N}_{\mathrm{f}} \alpha_{\mathrm{W}} = \mathrm{N}_{\mathrm{tf}}\alpha_{\mathrm{W}} \\ \nonumber \mathrm{a}_{\mathrm{x}}&=& \alpha_{\mathrm{W}} \, \mathrm{r}\\ \end{eqnarray}

Para um torque \(\mathrm{T}_{\mathrm{e}}\), qual é a força longitudinal \(\mathrm{F}_{\mathrm{x}}\)?

\(\mathrm{F}_{\mathrm{x}}=\frac{\mathrm{T}_{\mathrm{e}} \mathrm{N}_{\mathrm{tf}}}{\mathrm{r}}-\left\{\left(\mathrm{I}_{\mathrm{e}}+\mathrm{I}_{\mathrm{t}}\right) \mathrm{N}_{\mathrm{tf}}^{2}+\mathrm{I}_{\mathrm{d}} \mathrm{N}_{\mathrm{f}}^{2}+\mathrm{I}_{\mathrm{w}}\right\} \frac{\mathrm{a}_{\mathrm{x}}}{\mathrm{r}^{2}}\)

\(\mathrm{F}_{\mathrm{x}}=\frac{\mathrm{T}_{\mathrm{e}} \mathrm{N}_{\mathrm{tf}} \eta_{\mathrm{tf}}}{\mathrm{r}}-\left\{\left(\mathrm{I}_{\mathrm{e}}+\mathrm{I}_{\mathrm{t}}\right) \mathrm{N}_{\mathrm{tf}}^{2}+\mathrm{I}_{\mathrm{d}} \mathrm{N}_{\mathrm{f}}^{2}+\mathrm{I}_{\mathrm{w}}\right\} \frac{\mathrm{a}_{\mathrm{x}}}{\mathrm{r}^{2}}\)

Massa efetiva

\(\mathrm{F}_{\mathrm{x}}=\frac{\mathrm{T}_{\mathrm{e}} \mathrm{N}_{\mathrm{tf}} \eta_{\mathrm{tf}}}{\mathrm{r}}-\left\{\left(\mathrm{I}_{\mathrm{e}}+\mathrm{I}_{\mathrm{t}}\right) \mathrm{N}_{\mathrm{tf}}^{2}+\mathrm{I}_{\mathrm{d}} \mathrm{N}_{\mathrm{f}}^{2}+\mathrm{I}_{\mathrm{w}}\right\} \frac{\mathrm{a}_{\mathrm{x}}}{\mathrm{r}^{2}}\)

\(\mathrm{M} \mathrm{a}_{\mathrm{x}}=\frac{\mathrm{W}}{\mathrm{g}} \mathrm{a}_{\mathrm{x}}=\mathrm{F}_{\mathrm{X}}-\mathrm{R}_{\mathrm{X}}-\mathrm{D}_{\mathrm{A}}-\mathrm{R}_{\mathrm{hx}}-\mathrm{W} \sin \Theta\)

\((\mathrm{M} + \mathrm{M}_{\mathrm{r}}) \mathrm{a}_{\mathrm{x}}=\frac{(\mathrm{W}+\mathrm{W}_{\mathrm{r}})}{\mathrm{g}} \mathrm{a}_{\mathrm{x}}=\frac{\mathrm{T}_{\mathrm{e}} \mathrm{N}_{\mathrm{tf}} \eta_{\mathrm{tf}}}{\mathrm{r}}-\mathrm{R}_{\mathrm{X}}-\mathrm{D}_{\mathrm{A}}-\mathrm{R}_{\mathrm{hx}}-\mathrm{W} \sin \Theta\)

Massa equivalente dos elementos rotativos:

\(\mathrm{M}_{\mathrm{r}} = \frac{\left\{\left(\mathrm{I}_{\mathrm{e}}+\mathrm{I}_{\mathrm{t}}\right) \mathrm{N}_{\mathrm{tf}}^{2}+\mathrm{I}_{\mathrm{d}} \mathrm{N}_{\mathrm{f}}^{2}+\mathrm{I}_{\mathrm{w}}\right\} }{\mathrm{r}^{2}}\)

Massa efetiva: \(\mathrm{M} + \mathrm{M}_{\mathrm{r}}\)

Fator de massa: \(\frac{\mathrm{M} + \mathrm{M}_{\mathrm{r}}}{\mathrm{M}}\)

Escalonamento

\(\mathrm{\omega}_{\mathrm{e}} = \mathrm{RPM} \frac{2 \pi}{60}\)

\(\mathrm{\omega}_{\mathrm{w}} = \frac{\mathrm{\omega}_{\mathrm{e}}}{\mathrm{N}_{\mathrm{tf}}} \)

\(\mathrm{v}_{\mathrm{x}} = \mathrm{\omega}_{\mathrm{w}} \mathrm{r} \)

\(\mathrm{v}_{\mathrm{x}} = \frac{2 \pi}{60} \frac{\mathrm{r}}{\mathrm{N}_{\mathrm{tf}}} \mathrm{RPM} \)
